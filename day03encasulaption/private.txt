	}
	
	public int GetAge(string password) 
	{
		if (password == "pass")
		{
			return _catAge;
		}

		{
			return 0;
		}
	}
	public void SetAge(int age)
	{
		if (age > 0)
		{
			_catAge = age;
		} 
	}
